SENSOR DE MOVIMENTO

Esse é o nosso sensor de movimento e ele é relativamente fácil de montar e entender. Como o nome já diz, ele capta o movimento da pessoa ou objeto e o LED aciona na protoboard, e na ARDUINO IDE é mostrado no monitor serial "MOVIMENTO DETECTADO" ou "SEM MOVIMENTO". Bom, é mais fácil na prática! O código está aí, agora é só testar! 


#define SENSOR_PIR 2
#define LED 3

void setup() {

  pinMode(SENSOR_PIR, INPUT);
  pinMode(LED, OUTPUT);

  Serial.begin(9600);
}

void loop() {

  bool estado_sensor =
    digitalRead(SENSOR_PIR);

  if (estado_sensor == HIGH) {

    digitalWrite(LED, HIGH);

    Serial.println(
      "MOVIMENTO DETECTADO ✓"
    );

  } else {

    digitalWrite(LED, LOW);

    Serial.println(
      "SEM MOVIMENTO ☓"
    );

  }

  delay(200);
}

DETECTOR DE FUMAÇA

Este é o nosso detector de fumaça. Em comparação com o detector de movimento, ele é um pouco mais complexo de montar, mas nada impossível! O dispositivo detecta a presença de gases e fumaça no ambiente e aciona um alarme sonoro e visual. Caso seja detectado gás ou fumaça na área, um LED vermelho será aceso indicando: "⚠️ GÁS DETECTADO!" Se o ambiente estiver limpo, um LED verde acenderá com a mensagem: "✅ Ambiente Seguro" Observação: A imagem foi retirada de um site oficial. 


#define MQ_analogico A0
#define MQ_digital 4

#define Buzzer 5

#define Led_Verde 6
#define Led_Vermelho 7

int valor_analogico;
int valor_digital;

void setup() {

  Serial.begin(9600);

  pinMode(MQ_analogico, INPUT);
  pinMode(MQ_digital, INPUT);

  pinMode(Buzzer, OUTPUT);

  pinMode(Led_Verde, OUTPUT);
  pinMode(Led_Vermelho, OUTPUT);

  digitalWrite(Led_Verde, HIGH);
  digitalWrite(Led_Vermelho, LOW);

  noTone(Buzzer);
}

void loop() {

  valor_analogico =
    analogRead(MQ_analogico);

  valor_digital =
    digitalRead(MQ_digital);

  if (valor_analogico > 70) {

    digitalWrite(
      Led_Vermelho,
      HIGH
    );

    digitalWrite(
      Led_Verde,
      LOW
    );

    tone(Buzzer, 1000);

    Serial.println(
      "⚠️ GÁS DETECTADO!"
    );

  } else {

    digitalWrite(
      Led_Vermelho,
      LOW
    );

    digitalWrite(
      Led_Verde,
      HIGH
    );

    noTone(Buzzer);

    Serial.println(
      "✅ Ambiente Seguro"
    );

  }

  delay(200);
}
